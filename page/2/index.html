class Solution {
    public int maxProfit(int[] prices) {
        int preTotal = 0;
        for (int idxMin = 0; idxMin < prices.length; idxMin++) {
            int start = idxMin + 1;
            int count = prices.length - start;
            for (int i = 0; i < count; i++) {
                int total = 0;
                int remainLen = prices.length - start - i;
                int subArrayLen = remainLen-1;  // Exclude the next one
                if (subArrayLen > 0) {
                    int[] subArray = new int[subArrayLen];
                    System.arraycopy(prices, start+1+i, subArray, 0, subArrayLen);
                    total = prices[start+i] - prices[idxMin] + maxProfit(subArray);
                } else {
                    total = prices[start+i] - prices[idxMin];
                }

                if (total > preTotal) {
                    preTotal = total;
                }
            }
        }

        return preTotal; 
    }
}
